const startBtn = document.getElementById('startBtn'); const status = document.getElementById('status'); const progressBar = document.getElementById('progressBar'); const progressBarFill = document.getElementById('progressBarFill'); let videoFile = null; let srtFile = null; // 1. 初始化系統並加載核心 async function init() { status.innerText = '正在初始化 FFmpeg 核心與配置環境，請稍候...'; try { await ffmpeg.load(); status.innerText = '✅ 系統就緒！請先選取影片與字幕檔案。'; status.style.color = '#10b981'; } catch (e) { status.innerText = '❌ 核心載入失敗。\n請確保 coi-serviceworker.js 已推送到 GitHub 且與 index.html 同目錄。\n並請使用 Chrome/Edge 開啟。'; status.style.color = '#ef4444'; console.error(e); } } // 監聽檔案選擇 videoUploader.addEventListener('change', (e) => { videoFile = e.target.files[0]; if(videoFile) { document.getElementById('videoName').innerText = `已選取: ${videoFile.name}`; checkReady(); } }); srtUploader.addEventListener('change', (e) => { srtFile = e.target.files[0]; if(srtFile) { document.getElementById('srtName').innerText = `已選取: ${srtFile.name}`; checkReady(); } }); function checkReady() { if (videoFile && srtFile) { startBtn.disabled = false; progressBarFill.style.width = `${percentage}%`; status.innerText = `⏳ 正在努力壓制字幕中... ${percentage}% \n(請勿關閉網頁，影片越大需要越久時間)`; status.style.color = '#0284c7'; }); // 2. 開始轉檔 startBtn.addEventListener('click', async () => { if (!videoFile || !srtFile) return; startBtn.disabled = true; startBtn.classList.add('btn-disabled'); progressBar.style.display = 'block'; progressBarFill.style.width = '0%'; status.style.color = '#0f172a'; // 統一內部檔名 const videoIn = 'input' + videoFile.name.substring(videoFile.name.lastIndexOf('.')); const srtIn = 'subtitles.srt'; const fontName = 'font.ttf'; // 💡 虛擬字型檔名 const videoOut = 'output_with_subtitles.mp4'; try { // 💡 步驟 A：遠端抓取輕量化中文字型（使用 cdnjs 的思源黑體，確保瀏覽器有字型可用） status.innerText = '📡 正在從雲端載入中文字型系統（約 4MB，僅第一次需要）...'; // 這裡我們直接抓取繁體中文思源黑體子集 ffmpeg.FS('writeFile', fontName, await fetchFile('https://cdn.jsdelivr.net/npm/@fontsource/noto-sans-tc/files/noto-sans-tc-chinese-traditional-400-normal.woff')); // 如果上述 cdn 速度不理想，最穩健作法是下載一組 ttf 放在你 github 專案目錄，然後用 fetchFile('font.ttf') 載入。 // 為求保險，這裡改用一組通用的免費可用 ttf 字型位址： ffmpeg.FS('writeFile', fontName, await fetchFile('https://fastly.jsdelivr.net/gh/googlefonts/noto-fonts@main/hinted/ttf/NotoSans/NotoSans-Regular.ttf')); // 💡 最佳實踐：改用一組穩定、體積小且包含中文字元的字型： status.innerText = '📡 正在快取中文字體以便進行圖層渲染...'; ffmpeg.FS('writeFile', fontName, await fetchFile('https://fastly.jsdelivr.net/gh/orioncactus/pretendard@v1.3.9/packages/pretendard/dist/public/static/alternative/Pretendard-Bold.ttf')); status.style.color = '#ef4444'; console.error(err); } finally { startBtn.disabled = false; startBtn.classList.remove('btn-disabled'); } }); init();
